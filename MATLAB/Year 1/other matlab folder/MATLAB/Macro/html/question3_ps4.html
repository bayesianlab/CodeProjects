% Params
beta = 0.9;
gamma = 2;
z_2 = 1.10;
p_22 = 0.9;
power = 1- gamma;

% p_11 Defined
p_11 = [.2; .8];
z_1 =  [1.03; 1.10];

% Expected Value calculations, calculated using the stationary
% distribtuion and the given values for z.
expectedValues = [1.092; 1.0767];

% Diagonal Matrix
diag1 = [z_1(1,1)^power 0; 0 z_2^power];
diag2 = [z_1(2,1)^power 0; 0 z_2^power];

% Markov Transition Matrix
transitionMat1 = [p_11(1,1) 1-p_11(1,1); 1-p_22 p_22];
transitionMat2 = [p_11(2,1) 1-p_11(1,1); 1-p_22 p_22];

% All multiplied together created by funciton createMatrixG
matrixG11 = createMatrixG(transitionMat1, diag1, beta);
matrixG12 = createMatrixG(transitionMat1, diag2, beta);
matrixG21 = createMatrixG(transitionMat2, diag1, beta);
matrixG22 = createMatrixG(transitionMat2, diag2, beta);

% The only needed elements of above.
g11 = matrixG11(1,1);
g12 = matrixG12(1,1);
g21 = matrixG21(1,1);
g22 = matrixG22(1,1);

omega_11 = genOmega(expectedValues(1,1), g11, beta,...
    power);
omega_12 = genOmega(expectedValues(1,1), g12, beta,...
    power);
omega_21 = genOmega(expectedValues(2,1), g21, beta,...
    power);
omega_22 = genOmega(expectedValues(2,1), g22, beta,...
    power);
rowNames = {'z=1.03', 'z=1.10'};
colNames = {'p=0.2', 'p=0.8'};
OMEGA = [omega_11 omega_12; omega_21 omega_22]
ANSWERS = dataset({OMEGA rowNames{1} rowNames{2}}...
    ,'obsnames', {colNames{1} colNames{2}})
