385 template <typename Vector> PyObject *get_array(const Vector &v) {
386  detail::_interpreter::get(); // interpreter needs to be initialized for the
387  // numpy commands to work
388  // both Eigen::Matrix<..> and std::vector<..> have the member value_type
389  NPY_TYPES type = select_npy_type<typename Vector::value_type>::type;
390  if (type == NPY_NOTYPE) {
391  std::vector<double> vd(v.size());
392  npy_intp vsize = v.size();
393  // Eigen Vectors do not support begin/end() in the currently stable version
394  // this can be changed once Eigen 3.4. is released
395  // data() returns a pointer to the storage of the first element. If the
396  // vector is modified afterwards, it may be rendedered invalid.
397  // Note, that this is not an issue since get_array() is called by a
398  // plot command using the instantaneous state of the vector.
399  // The pointer is not reused. If the vector is plotted anew, data() is
400  // called again and again get's the current, valid storage location.
401  std::copy(v.data(), v.data() + v.size(), vd.begin());
402  PyObject *varray =
403  PyArray_SimpleNewFromData(1, &vsize, NPY_DOUBLE, (void *)(vd.data()));
404  return varray;
405  }
406 
407  npy_intp vsize = v.size();
408  PyObject *varray =
409  PyArray_SimpleNewFromData(1, &vsize, type, (void *)(v.data()));
410  return varray;
411 }

413 // specialized get_2darray function for nested std::vectors
414 template <typename Numeric>
415 PyObject *get_2darray(const std::vector<::std::vector<Numeric>> &v) {
416  detail::_interpreter::get(); // interpreter needs to be initialized for the
417  // numpy commands to work
418  if (v.size() < 1)
419  throw std::runtime_error("get_2darray v too small");
420 
421  npy_intp vsize[2] = {static_cast<npy_intp>(v.size()),
422  static_cast<npy_intp>(v[0].size())};
423 
424  PyArrayObject *varray =
425  (PyArrayObject *)PyArray_SimpleNew(2, vsize, NPY_DOUBLE);
426 
427  double *vd_begin = static_cast<double *>(PyArray_DATA(varray));
428 
429  for (const ::std::vector<Numeric> &v_row : v) {
430  if (v_row.size() != static_cast<size_t>(vsize[1]))
431  throw std::runtime_error("mismatched array size");
432  std::copy(v_row.begin(), v_row.end(), vd_begin);
433  vd_begin += vsize[1];
434  }
435 
436  return reinterpret_cast<PyObject *>(varray);
437 }

440 template <typename Matrix> PyObject *get_2darray(const Matrix &A) {
441  detail::_interpreter::get(); // interpreter needs to be initialized for the
442  // numpy commands to work
443  if (A.size() < 1)
444  throw std::runtime_error("get_2darray A too small");
445 
446  npy_intp vsize[2] = {static_cast<npy_intp>(A.rows()),
447  static_cast<npy_intp>(A.cols())};
448 
449  PyArrayObject *varray =
450  (PyArrayObject *)PyArray_SimpleNew(2, vsize, NPY_DOUBLE);
451 
452  double *vd_begin = static_cast<double *>(PyArray_DATA(varray));
453 
454  for (std::size_t i = 0; i < A.rows(); ++i) {
455  for (std::size_t j = 0; j < A.cols(); ++j) {
456  *(vd_begin + i * A.cols() + j) = A(i, j);
457  }
458  }
459 
460  return reinterpret_cast<PyObject *>(varray);
461 }

477 // @brief Since most of the plot commands require the exact same usage apart
478 // from the call to the correct Python function, we encapsulate this
479 // @param pyfunc The matplotlib function to be called with the given arguments
480 // @param x The x vector

483 // @param keywords Additional keywords, such as label
484 // @return true if plot was successful, false otherwise
485 template <typename VectorX, typename VectorY>
486 bool plot_base(PyObject *const pyfunc, const VectorX &x, const VectorY &y,
487  const std::string &s = "",
488  const std::map<std::string, std::string> &keywords = {}) {
489  assert(x.size() == y.size());
490 
491  PyObject *xarray = get_array(x);
492  PyObject *yarray = get_array(y);
493 
494  PyObject *pystring = PyString_FromString(s.c_str());
495 
496  PyObject *plot_args = PyTuple_New(3);
497  PyTuple_SetItem(plot_args, 0, xarray);
498  PyTuple_SetItem(plot_args, 1, yarray);
499  PyTuple_SetItem(plot_args, 2, pystring);
500 
501  PyObject *kwargs = PyDict_New();
502  for (auto const &item : keywords) {
503  PyDict_SetItemString(kwargs, item.first.c_str(),
504  PyString_FromString(item.second.c_str()));
505  }
506 
507  PyObject *res = PyObject_Call(pyfunc, plot_args, kwargs);
508 
509  Py_DECREF(plot_args);
510  Py_DECREF(kwargs);
511  if (res)
512  Py_DECREF(res);
513 
514  return res;
515 }

701 // @brief plot_surface for datapoints (x_ij, y_ij, z_ij) with i,j = 0..n
702 // @param x The x values of the datapoints in a matrix
703 // @param y The y values of the datapoints in a matrix
704 // @param z The function value of the datapoints in a matrix
705 // @param keywords Additional keywords
706 template <typename Matrix>
707 void plot_surface(const Matrix &x, const Matrix &y, const Matrix &z,
708  const std::map<std::string, std::string> &keywords =
709  std::map<std::string, std::string>()) {
710  // We lazily load the modules here the first time this function is called
711  // because I'm not sure that we can assume "matplotlib installed" implies
712  // "mpl_toolkits installed" on all platforms, and we don't want to require
713  // it for people who don't need 3d plots.
714  static PyObject *mpl_toolkitsmod = nullptr, *axis3dmod = nullptr;
715  if (!mpl_toolkitsmod) {
716  detail::_interpreter::get();
717 
718  PyObject *mpl_toolkits = PyString_FromString("mpl_toolkits");
719  PyObject *axis3d = PyString_FromString("mpl_toolkits.mplot3d");
720  if (!mpl_toolkits || !axis3d) {
721  throw std::runtime_error("couldnt create string");
722  }
723 
724  mpl_toolkitsmod = PyImport_Import(mpl_toolkits);
725  Py_DECREF(mpl_toolkits);
726  if (!mpl_toolkitsmod) {
727  throw std::runtime_error("Error loading module mpl_toolkits!");
728  }
729 
730  axis3dmod = PyImport_Import(axis3d);
731  Py_DECREF(axis3d);
732  if (!axis3dmod) {
733  throw std::runtime_error("Error loading module mpl_toolkits.mplot3d!");
734  }
735  }
736 
737  assert(x.size() == y.size());
738  assert(y.size() == z.size());
739 
740  // using numpy arrays
741  PyObject *xarray = get_2darray(x);
742  PyObject *yarray = get_2darray(y);
743  PyObject *zarray = get_2darray(z);
744 
745  // construct positional args
746  PyObject *args = PyTuple_New(3);
747  PyTuple_SetItem(args, 0, xarray);
748  PyTuple_SetItem(args, 1, yarray);
749  PyTuple_SetItem(args, 2, zarray);
750 
751  // Build up the kw args.
752  PyObject *kwargs = PyDict_New();
753  PyDict_SetItemString(kwargs, "rstride", PyInt_FromLong(1));
754  PyDict_SetItemString(kwargs, "cstride", PyInt_FromLong(1));
755 
756  PyObject *python_colormap_coolwarm = PyObject_GetAttrString(
757  detail::_interpreter::get().s_python_colormap, "coolwarm");
758 
759  PyDict_SetItemString(kwargs, "cmap", python_colormap_coolwarm);
760 
761  for (std::map<std::string, std::string>::const_iterator it = keywords.begin();
762  it != keywords.end(); ++it) {
763  PyDict_SetItemString(kwargs, it->first.c_str(),
764  PyString_FromString(it->second.c_str()));
765  }
766 
767  PyObject *fig =
768  PyObject_CallObject(detail::_interpreter::get().s_python_function_figure,
769  detail::_interpreter::get().s_python_empty_tuple);
770  if (!fig)
771  throw std::runtime_error("Call to figure() failed.");
772 
773  PyObject *gca_kwargs = PyDict_New();
774  PyDict_SetItemString(gca_kwargs, "projection", PyString_FromString("3d"));
775 
776  PyObject *gca = PyObject_GetAttrString(fig, "gca");
777  if (!gca)
778  throw std::runtime_error("No gca");
779  Py_INCREF(gca);
780  PyObject *axis = PyObject_Call(
781  gca, detail::_interpreter::get().s_python_empty_tuple, gca_kwargs);
782 
783  if (!axis)
784  throw std::runtime_error("No axis");
785  Py_INCREF(axis);
786 
787  Py_DECREF(gca);
788  Py_DECREF(gca_kwargs);
789 
790  PyObject *plot_surface = PyObject_GetAttrString(axis, "plot_surface");
791  if (!plot_surface)
792  throw std::runtime_error("No surface");
793  Py_INCREF(plot_surface);
794  PyObject *res = PyObject_Call(plot_surface, args, kwargs);
795  if (!res)
796  throw std::runtime_error("failed surface");
797  Py_DECREF(plot_surface);
798 
799  Py_DECREF(axis);
800  Py_DECREF(args);
801  Py_DECREF(kwargs);
802  if (res)
803  Py_DECREF(res);
804 }
